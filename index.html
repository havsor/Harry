const fs = require('fs'); const path = require('path'); const XLSX = require('xlsx'); const WebSocket = require('ws'); const filePath = path.join(__dirname, 'data.xlsx'); // Excel-filens plassering let previousData = null; // Opprett WebSocket-server const wss = new WebSocket.Server({ port: 8080 }); // Funksjon for å lese Excel-filen og sende data til en klient function readExcelData() { const workbook = XLSX.readFile(filePath); const sheetName = workbook.SheetNames[0]; const worksheet = workbook.Sheets[sheetName]; const data = XLSX.utils.sheet_to_json(worksheet, { header: 1 }); previousData = data; } // Initialiser data når serveren starter readExcelData(); // Send data til alle tilkoblede klienter function broadcastExcelData() { wss.clients.forEach(client => { if (client.readyState === WebSocket.OPEN) { client.send(JSON.stringify(previousData)); } }); } // Når en ny klient kobler til, send de siste dataene umiddelbart wss.on('connection', (ws) => { // Send eksisterende data til ny klient if (previousData) { ws.send(JSON.stringify(previousData)); } }); // Les og send oppdateringer til alle klienter ved filendringer fs.watchFile(filePath, { interval: 1000 }, () => { const workbook = XLSX.readFile(filePath); const sheetName = workbook.SheetNames[0]; const worksheet = workbook.Sheets[sheetName]; const newData = XLSX.utils.sheet_to_json(worksheet, { header: 1 }); // Sjekk om dataene har endret seg if (JSON.stringify(newData) !== JSON.stringify(previousData)) { previousData = newData; broadcastExcelData(); // Send oppdaterte data til alle klienter } }); console.log('WebSocket-server kjører på ws://localhost:8080');
